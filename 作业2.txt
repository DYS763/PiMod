package tutorial;
import basemod.interfaces.*;
import com.evacipated.cardcrawl.modthespire.lib.SpireInitializer;
import com.megacrit.cardcrawl.cards.DamageInfo;
import com.megacrit.cardcrawl.rooms.AbstractRoom;

import basemod.BaseMod;
@SpireInitializer
public class Tutorial implements OnPlayerDamagedSubscriber,PostBattleSubscriber {
    public Tutorial() {
        BaseMod.subscribe(this);
    }

    public static void initialize() {
        new Tutorial();
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
    }
    public static int sum=0;
    @Override
    public int receiveOnPlayerDamaged(int damage, DamageInfo damageInfo) {
        //当玩家受到伤害，该函数接受玩家受到的伤害和伤害种类
        sum+=damage;
        return damage;
    }
    @Override
    public void receivePostBattle(AbstractRoom r) {
        //该函数中的内容将在战斗结束后执行一次
        System.out.println(sum);
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        System.out.print("hello world! 5*6+30=");
        sum=0;
    }

}
